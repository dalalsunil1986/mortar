    public module: Object source

    The module object used to resolve lazy requires.

    Example:

    import Context from 'mortar'
import assert from 'assert'

const context = new Context(module);
assert.strictEqual(module, context.module);
context.require('./models/User').as.singleton('CurrentUser');
context.retrieve('CurrentUser'); //lazily required the User model relative to this module

    public parent: Context source

    The parent Context instance. Allows to set up hierarchical dependency injection containers.

    Example:

    import Context from 'mortar'
import assert from 'assert'
const context = Context.create();
const child = context.spawn();
assert.strictEqual(child.parent, context);

    public require(id: String): object source

    Sets up lazy requireing of id when key is requested, relative to module.

    See Context#wire

    Params:

    NameTypeAttributeDescription
    id String

    the module id, e.g. "lodash" or "./models/User"

    Return:

    object

    Return Properties:

    NameTypeAttributeDescription
    as.singleton function(key: string): Context

    registers id to be lazily required as a singleton provider for key
    as.producer function(key: string): Context

    registers id to be lazily required as a producer for key
    as.value function(key: string): Context

    registers id to be lazily required as a value for key

    Example:

    import Context from 'mortar'

const context = Context.create(module);
context.require('./models/User').as.singleton('CurrentUser');
// no code has been loaded yet
context.retrieve('CurrentUser');
// now './models/User' is resolved and used to produce a singleton instance

    public resolve(subject: Function, constructable: Boolean): * source

    Retrieves subject's dependencies and calls it.

    Params:

    NameTypeAttributeDescription
    subject Function
    constructable Boolean
    • optional
    • default: false

    whether new should be used or the function should be called directly

    Return:

    *

    Example:

    import Context from 'mortar'

function qux(){
  return 'and thanks for all the fish!';
}
const context = Context.create();
context
  .wire('So long').as.value('foo')
  .wire(qux).as.producer('qux');

context.resolve(function (foo, qux){
  console.log(foo, qux); // outputs: So long, and thanks for all the fish!
});

    public using(contextOrMap: Context | Object): {resolve: resolve} source

    Allows you to override registered dependencies through another Context or mapping.

    Params:

    NameTypeAttributeDescription
    contextOrMap Context | Object

    Return:

    {resolve: resolve}

    Example:

    import Context from 'mortar'

const context = Context.create();
context
  .wire('So long').as.value('foo')
  .wire('and thanks for all the fish!').as.value('qux');

const subject = function (foo, qux){
  console.log(foo, qux);
);

context.using({
  foo: 'Adieu'
}).resolve(subject);
//outputs: Adieu, and thanks for all the fish!

    import Context from 'mortar'
